$(function(){
  $('#tree1').bootstrapTree({
    openOnLoad: true, // 默认打开树
    icons: { // 节点图标
      openedNodeIcon: 'glyphicon-folder-open',
      closedNodeIcon: 'glyphicon-folder-close',
      leafNodeIcon: 'glyphicon-file'
    }
  });
});
function closeTree(){ // 关闭
  $('#tree1').bootstrapTree('closeAll');
}
function openTree(){ // 打开
  $('#tree1').bootstrapTree('openAll');
}
function destroy(){ // 销毁
  $('#tree1').bootstrapTree('destroy');
}
function init(){ // 初始化
  $('#tree1').bootstrapTree('init');
}
