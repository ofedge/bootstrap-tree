省市id为从0递增序号, 区域id为从1开始不重复序号
用法 
      html:
      <select id="province"></select>
      <select id="city"></select>
      <select id="district"></select>
      
      js:
      $.provinceAndCity({
          province: '#province',
          city: '#city',
          district: '#district',
          provinceValue: '台湾',
          cityValue: '台湾',
          districtValue: '基隆市',
          textValue: false // 是否使用文本值作为选择的value, 即使用省市区名称作为value值
      });
